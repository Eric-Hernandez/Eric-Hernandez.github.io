int matrix_multiply (double* m1,int m1_rows,int m1_columns,double* m2,
    int m2_rows,int m2_columns,double* m3) {

      int i,j,k;

      if (m1_columns != m2_rows) {
        return (0);
      }

      for(i=0; i < m1_rows; i++) {
        for(j=0; j < m2_columns; j++) {
          *((m3+i*m2_columns)+j)=0;
          for(k=0; k < m2_rows; k++) {
            *((m3+i*m2_columns)+j) +=
            (*(m1+i*m1_columns+k)*(*(m2+k*m2_columns)+j));
          }
        }
      }

      return(1);
  }
